    const centerY = rect.top + rect.height / 2;
    const deltaX = (e.clientX - centerX) / rect.width;
    const deltaY = (e.clientY - centerY) / rect.height;

    const photos = gallery.querySelectorAll('.parallax-photo');

      const speed = 10 + i * 2;
      const moveX = deltaX * speed;
      const moveY = deltaY * speed;
      p.style.transform = `translate3d(${moveX}px, ${moveY}px, 0)`;
    });

  /* Animação partículas douradas TELA1 */
  const ctx = particlesCanvas.getContext('2d');
  const w = window.innerWidth;
  const h = window.innerHeight;
  particlesCanvas.width = w;
  particlesCanvas.height = h;

  class Particle {
    constructor(){
      this.x = Math.random() * w;
      this.y = Math.random() * h;
      this.radius = Math.random() * 1.5 + 0.5;
      this.speedY = Math.random()*0.2 + 0.05;
      this.opacity = Math.random();
      this.color = `rgba(191, 160, 89, ${this.opacity})`;
    }
    update(){
      this.y -= this.speedY;
      if(this.y < 0) {
        this.y = h;
        this.x = Math.random() * w;
      }
      this.opacity += (Math.random() - 0.5)*0.05;
      this.opacity = Math.min(Math.max(this.opacity, 0.1), 0.9);
      this.color = `rgba(191, 160, 89, ${this.opacity})`;
    }
    draw(){
      ctx.beginPath();
      ctx.fillStyle = this.color;
      ctx.shadowColor = 'rgba(191,160,89,0.8)';
      ctx.shadowBlur = 6;
      ctx.arc(this.x, this.y, this.radius, 0, Math.PI*2);
      ctx.fill();
    }
  }

  const particles = [];
  for(let i=0; i<50; i++) particles.push(new Particle());
  function animateParticles() {
    ctx.clearRect(0,0,w,h);
    particles.forEach(p => {
      p.update();
      p.draw();
    });
    requestAnimationFrame(animateParticles);
  }
  animateParticles();

  /* Partículas na revelação principal */
  const ctxReveal = particlesRevealCanvas.getContext('2d');
  particlesRevealCanvas.width = window.innerWidth;
  particlesRevealCanvas.height = window.innerHeight;

  class RevealParticle {
    constructor() {
      this.x = Math.random() * window.innerWidth;
      this.y = Math.random() * window.innerHeight;
      this.size = Math.random() * 10 + 5;
      this.speedY = 0.5 + Math.random();
      this.color = Math.random() > 0.7 ? 'rgba(251,191,187,0.9)' : 'rgba(191,160,89, 0.8)';
      this.type = Math.random() > 0.52 ? 'heart' : 'circle';
      this.angle = 0;
      this.angleSpeed = 0.05 + Math.random()*0.02;
    }
    update() {
      this.y += this.speedY;
      this.angle += this.angleSpeed;
      if(this.y > window.innerHeight + this.size) {
        this.y = -20;
        this.x = Math.random() * window.innerWidth;
      }
    }
    draw() {
      ctxReveal.save();
      ctxReveal.translate(this.x, this.y);
      if(this.type === 'heart') {
        ctxReveal.rotate(Math.sin(this.angle));
        ctxReveal.fillStyle = 'rgba(251,191,187,0.85)';
        ctxReveal.beginPath();
        const size = this.size / 5;
        ctxReveal.moveTo(0, -size);
        ctxReveal.bezierCurveTo(size, -2*size, 3*size, size/2, 0, 3*size);
        ctxReveal.bezierCurveTo(-3*size, size/2, -size, -2*size, 0, -size);
        ctxReveal.fill();
      } else {
        ctxReveal.fillStyle = this.color;
        ctxReveal.shadowColor = this.color;
        ctxReveal.shadowBlur = 10;
        ctxReveal.beginPath();
        ctxReveal.arc(0, 0, this.size / 3, 0, Math.PI * 2);
        ctxReveal.fill();
      }
      ctxReveal.restore();
    }
  }

  const revealParticles = [];
  for(let i=0; i<50; i++) revealParticles.push(new RevealParticle());

  function startRevealParticles() {
    function animateReveal() {
      ctxReveal.clearRect(0, 0, window.innerWidth, window.innerHeight);
      revealParticles.forEach(p => {
        p.update();
        p.draw();
      });
      requestAnimationFrame(animateReveal);
    }
    animateReveal();
  }

    particlesCanvas.width = window.innerWidth;
    particlesCanvas.height = window.innerHeight;
    particlesRevealCanvas.width = window.innerWidth;
    particlesRevealCanvas.height = window.innerHeight;
